%%  Enabling page breaks in math environment

%%  I, II, III,...
\newcounter{prob}
\newcommand{\problem}[1]{
    \vspace{4mm} 
    \noindent 
    \setcounter{prob}{#1} 
    \textbf{\Roman{prob}}.
}

%%  (1), (2), (3),...
\newcounter{que}
\newcommand{\question}[1]{
    \vspace{4mm} 
    \noindent 
    \setcounter{que}{#1} 
    \textbf{(\theque)}
}

%%  Answer
\newenvironment{answer}{
    \begin{tcolorbox}[
        sharp corners, 
        colback=red!10, 
        colframe=white, 
        breakable=true
    ] 
    \textbf{答案} 
    \hspace{1pt} 
    \setlength{\parindent}{10pt}
}{
    \end{tcolorbox}
}

%%  Supplement
\newenvironment{supplement}{
    \begin{tcolorbox}[
        sharp corners, 
        colback=blue!10, 
        colframe=white, 
        breakable=true
    ] 
    \textbf{補足} 
    \hspace{1pt} 
    \setlength{\parindent}{10pt}
}{
    \end{tcolorbox}
}
